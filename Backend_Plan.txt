⚠  Every test listed here maps directly to a real risk. If a test fails, a student either can't start their lab, loses their work, or gets billed for a container that never cleaned up.

Part D: Core Services
D.1  Auth Service
Handles JWT issuance for API access and separate short-lived tokens for terminal embeds. Uses two signing keys — RS256 for API tokens (verifiable without secret), HS256 for embed tokens (faster, symmetric).

// src/services/auth.service.ts (key functions)

export async function issueTokenPair(userId: string) {
  const accessToken = await new SignJWT({ sub: userId, type: 'access' })
    .setProtectedHeader({ alg: 'HS256' })
    .setExpirationTime('15m')
    .setIssuedAt()
    .sign(ACCESS_SECRET);

  const refreshToken = await new SignJWT({ sub: userId, type: 'refresh', jti: nanoid() })
    .setProtectedHeader({ alg: 'HS256' })
    .setExpirationTime('7d')
    .setIssuedAt()
    .sign(REFRESH_SECRET);

  // Store refresh token JTI in Redis for revocation
  await redis.setex(`refresh:${userId}:${jti}`, 604800, '1');
  return { accessToken, refreshToken };
}

export async function issueEmbedToken(sessionId: string, userId: string) {
  // Embed token is SCOPED — cannot be used for API calls
  return new SignJWT({ sub: userId, session: sessionId, type: 'embed' })
    .setProtectedHeader({ alg: 'HS256' })
    .setExpirationTime('10m')
    .setIssuedAt()
    .sign(EMBED_SECRET);  // Different secret from API tokens
}

export async function verifyAccessToken(token: string) {
  const { payload } = await jwtVerify(token, ACCESS_SECRET);
  if (payload.type !== 'access') throw new UnauthorizedError();
  return payload;
}

D.2  Lab Orchestrator Service
The heart of the system. Manages the full provision → ready → destroy lifecycle. Uses a BullMQ queue to prevent blocking the API thread during container startup.

// src/services/orchestrator.service.ts

export async function provisionLab(userId: string, labId: string) {
  // 1. Check quota
  const activeCount = await redis.scard(`user:${userId}:active_sessions`);
  if (activeCount >= MAX_SESSIONS_PER_USER) throw new QuotaExceededError();

  // 2. Check for existing session (idempotency)
  const existing = await db.query.sessions.findFirst({
    where: and(eq(sessions.userId, userId), eq(sessions.labId, labId),
               inArray(sessions.status, ['provisioning', 'ready']))
  });
  if (existing) throw new LabAlreadyRunningError(existing.id);

  // 3. Acquire provision lock (prevent race condition)
  const lockKey = `provision:lock:${userId}:${labId}`;
  const locked = await redis.set(lockKey, '1', 'EX', 30, 'NX');
  if (!locked) throw new LabAlreadyRunningError();

  // 4. Create session record
  const session = await db.insert(sessions).values({
    userId, labId, status: 'provisioning',
    ttlSeconds: lab.ttlSeconds,
    expiresAt: new Date(Date.now() + lab.ttlSeconds * 1000)
  }).returning();

  // 5. Add to provision queue (non-blocking)
  await provisionQueue.add('provision', { sessionId: session.id, labId, userId });

  return session;
}

// BullMQ Worker — runs in separate process
provisionWorker.process(async (job) => {
  const { sessionId, labId, userId } = job.data;
  const lab = await getLab(labId);

  // 1. Start container
  const subdomain = `sess-${nanoid(8).toLowerCase()}`;
  const container = await containerService.start({
    image: lab.dockerImage,
    subdomain,
    cpuLimit: lab.cpuLimit,
    memoryLimit: lab.memoryLimitMb * 1024 * 1024,
    labels: { 'lab.session': sessionId, 'lab.user': userId }
  });

  // 2. Run init script if defined
  if (lab.initScript) {
    await containerService.exec(container.id, lab.initScript, { timeout: 30000 });
  }

  // 3. Wait for ttyd to be ready (health check)
  await waitForPort(subdomain, lab.exposePort, 30000);

  // 4. Issue embed token, build terminal URL
  const embedToken = await authService.issueEmbedToken(sessionId, userId);
  const terminalUrl = `https://${subdomain}.${BASE_DOMAIN}/terminal?token=${embedToken}`;

  // 5. Register Traefik route
  await traefikService.registerRoute(subdomain, container.id);

  // 6. Update session to ready
  await db.update(sessions).set({
    status: 'ready', containerId: container.id,
    subdomain, terminalUrl, readyAt: new Date()
  }).where(eq(sessions.id, sessionId));

  // 7. Set Redis heartbeat key
  await redis.setex(`session:${sessionId}:alive`, lab.ttlSeconds, '1');
  await redis.sadd(`user:${userId}:active_sessions`, sessionId);

  // 8. Emit WebSocket event to frontend
  await eventService.emit(sessionId, { type: 'session_ready', terminalUrl });
});

export async function startContainer(opts: ContainerOpts) {

    Image: opts.image,
    name: `lab-${opts.subdomain}`,

    Labels: {
      'traefik.enable': 'true',
      [`traefik.http.routers.${opts.subdomain}.rule`]: `Host("${opts.subdomain}.${BASE_DOMAIN}")`,
      [`traefik.http.services.${opts.subdomain}.loadbalancer.server.port`]: `${opts.port}`,
      'lab.managed': 'true',
      'lab.session': opts.sessionId,
      'lab.user': opts.userId,
    },
    HostConfig: {
      NanoCpus: Math.floor(opts.cpuLimit * 1e9),
      Memory: opts.memoryLimit,
      NetworkMode: CONTAINER_NETWORK,
      AutoRemove: false,    // We manage cleanup explicitly
      ReadonlyRootfs: false,
      CapDrop: ['ALL'],     // Drop all capabilities
      CapAdd: ['CHOWN', 'SETUID', 'SETGID'],  // Only what's needed
      SecurityOpt: ['no-new-privileges:true'],
    }
  });

  await container.start();
  return container;
}

export async function execInContainer(containerId: string, cmd: string, timeoutMs = 10000) {
  const container = docker.getContainer(containerId);
  const exec = await container.exec({
    Cmd: ['/bin/sh', '-c', cmd],
    AttachStdout: true,
    AttachStderr: true,
  });

  return Promise.race([
    exec.start({ hijack: true, stdin: false }),
    new Promise((_, rej) => setTimeout(() => rej(new ValidationTimeoutError()), timeoutMs))
  ]);
}

D.4  Session Service (Heartbeat & Lifecycle)
// src/services/session.service.ts

export async function heartbeat(sessionId: string, userId: string) {
  // Verify ownership
  const session = await getSessionOrThrow(sessionId, userId);

  // Reset Redis TTL
  const ttl = session.ttlSeconds;
  await redis.expire(`session:${sessionId}:alive`, ttl);

  // Update DB expires_at
  const newExpiry = new Date(Date.now() + ttl * 1000);
  await db.update(sessions).set({ expiresAt: newExpiry })
    .where(eq(sessions.id, sessionId));

  return { timeRemainingSeconds: ttl, status: session.status };
}

// Redis keyspace notification triggers this when session key expires
export async function onSessionExpired(sessionId: string) {
  const session = await db.query.sessions.findFirst({
    where: eq(sessions.id, sessionId)
  });
  if (!session || session.status !== 'ready') return;

  // Destroy container
  await containerService.destroyContainer(session.containerId);
  await traefikService.removeRoute(session.subdomain);

  // Update DB
  await db.update(sessions).set({ status: 'expired', endedAt: new Date() })
    .where(eq(sessions.id, sessionId));

  // Clean up Redis
  await redis.srem(`user:${session.userId}:active_sessions`, sessionId);

  // Notify frontend
  await eventService.emit(sessionId, { type: 'session_expired', reason: 'ttl_reached' });
}

F.1  Architecture
The WS gateway uses Redis pub/sub so that events emitted from BullMQ workers (which run in separate processes) reach the client's WS connection on the API server. One Redis channel per session.

// src/websocket/gateway.ts

fastify.register(async (fastify) => {
  fastify.get('/ws/v1/labs/:sessionId/events', { websocket: true },
    async (socket, req) => {
      const { sessionId } = req.params;

      // Authenticate via subprotocol token
      const token = req.headers['sec-websocket-protocol']?.split(', ')[1];
      const user = await authService.verifyAccessToken(token).catch(() => null);
      if (!user) return socket.close(4001, 'Unauthorized');

      // Verify session ownership
      const session = await getSession(sessionId);
      if (!session || session.userId !== user.sub) return socket.close(4003, 'Forbidden');

      // Subscribe to Redis channel
      const subscriber = redis.duplicate();
      await subscriber.subscribe(`ws:channel:${sessionId}`);

      subscriber.on('message', (channel, message) => {
        if (socket.readyState === WebSocket.OPEN) socket.send(message);
      });

      socket.on('close', () => subscriber.unsubscribe().then(() => subscriber.quit()));
    }
  );
});

// From any service or worker — emit event to all connected clients
export async function emitEvent(sessionId: string, event: LabEvent) {
  await redis.publish(`ws:channel:${sessionId}`, JSON.stringify(event));
}

Part G: Resource Monitoring
G.1  Stats Collection
A BullMQ repeating job polls Docker stats every 10 seconds for all active containers. Docker's stats API streams CPU and memory data. We store snapshots in PostgreSQL and cache the latest in Redis for low-latency API reads.

// src/jobs/monitor.job.ts

monitorWorker.process(async () => {
  const activeSessions = await db.query.sessions.findMany({
    where: eq(sessions.status, 'ready'),
    columns: { id: true, containerId: true, userId: true }
  });

  await Promise.all(activeSessions.map(async (session) => {
    const stats = await docker.getContainer(session.containerId).stats({ stream: false });

    // Calculate CPU % from Docker stats delta
    const cpuDelta = stats.cpu_stats.cpu_usage.total_usage
                   - stats.precpu_stats.cpu_usage.total_usage;
    const systemDelta = stats.cpu_stats.system_cpu_usage
                      - stats.precpu_stats.system_cpu_usage;
    const cpuPercent = (cpuDelta / systemDelta) * stats.cpu_stats.online_cpus * 100;

    const memoryMb = stats.memory_stats.usage / (1024 * 1024);

    // Store snapshot
    await db.insert(resourceSnapshots).values({
      sessionId: session.id, cpuPercent, memoryMb
    });

    // Cache latest
    await redis.setex(`resource:${session.id}:latest`, 30,
      JSON.stringify({ cpuPercent, memoryMb, recordedAt: new Date() }));

    // Warn on high CPU
    if (cpuPercent > 90) {
      await emitEvent(session.id, {
        type: 'resource_warning', cpu_percent: cpuPercent, memory_mb: memoryMb
      });
    }
  }));
});

// src/services/scheduler.service.ts

export async function bookSlot(userId: string, labId: string,
                                slotStart: Date, slotEnd: Date) {
  // 1. Validate time (not in past, within booking window)
  if (slotStart <= new Date()) throw new InvalidScheduleTimeError();
  if (slotStart > addDays(new Date(), 14)) throw new TooFarInFutureError();

  // 2. Check user overlap
  const userOverlap = await db.query.bookings.findFirst({
    where: and(
      eq(bookings.userId, userId),
      inArray(bookings.status, ['confirmed']),
      lte(bookings.slotStart, slotEnd),
      gte(bookings.slotEnd, slotStart)
    )
  });
  if (userOverlap) throw new SlotOverlapError();

  // 3. Acquire slot lock (prevents race condition on booking)
  const lockKey = `booking:lock:${labId}:${slotStart.getTime()}`;
  const locked = await redis.set(lockKey, userId, 'EX', 10, 'NX');

  if (!locked) {
    // Slot being booked concurrently — add to queue
    const queuePos = await addToQueue(userId, labId, slotStart, slotEnd);
    return { status: 'queued', queuePosition: queuePos };
  }

  try {
    // 4. Create booking
    const booking = await db.insert(bookings).values({
      userId, labId, slotStart, slotEnd, status: 'confirmed'
    }).returning();

    // 5. Schedule auto-provision job
    await scheduleQueue.add('auto-provision',
      { bookingId: booking.id, userId, labId },
      { delay: slotStart.getTime() - Date.now() - 5000 }  // 5s before start
    );

    // 6. Schedule 15-min reminder
    const reminderDelay = slotStart.getTime() - Date.now() - (15 * 60 * 1000);
    if (reminderDelay > 0) {
      await notificationQueue.add('reminder',
        { bookingId: booking.id, userId },
        { delay: reminderDelay }
      );
    }

    return { status: 'confirmed', bookingId: booking.id };
  } finally {
    await redis.del(lockKey);
  }
}
